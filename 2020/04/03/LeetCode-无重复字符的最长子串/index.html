LeetCode-无重复字符的最长子串
Created 2020-04-03| Updated 2020-04-03
Post View:

  首先想到用哈希表，设置一个与26个字母映射的哈希表，读取string的字符，每次只要表中某个元素次数为2，就将表清空，然后此元素置1.
哈希表用的不是很熟，用vector实现了一下。
[TOC]
#include
#include
#include
using namespace std;

int lengthOfLongestSubstring(string s) {
    int len=s.length();
    if(len==0)return 0;
    vector<char>vec;
    vector<char>temp;
    vec.push_back(s[0]);
    int ans=vec.size();
    int n=0;

    for(int i=1;i<len;i++)
    {
        for(int j=0;j<vec.size();j++)
        {
            n=vec.size();
            if(ans<n)ans=n;
            if(s[i]==vec[j])
            {
                vec.clear();
                //vec=temp;
                //vec.push_back(s[i]);
                break;
            }
        } 
        vec.push_back(s[i]);
    }
    int m=vec.size();
    if(ans<m)ans=m;
    return ans;
}

  提交发现测试用例只能通过一半，卡在了如’dvdf’的用例，这才意识到上述思路是有漏洞的，问题出在：当检测到string字符与vector字符相同时就清空vector这是不对的.正确思路应该是删除掉vector中当前字符以及当前字符以前的元素。可以使用迭代器和erase()函数实现。

##暴力算法
  逐个检查所有的子字符串，看它是否不含有重复的字符。设置一个bool类型函数来判断子字符串中的字符是否都是唯一的。然后遍历该字符串的所有可能的子字符串。记录最大值。所有可能子字符串用两个for循环完成，判断字符是否唯一可以使用容器完成。时间复杂度超时。
[TOC]
#include
#include
using namespace std;
class Solution {
public:
bool CheckSbuStr(string str,int start,int end)
{
vectorvec;
vector::iterator it;
for(int i=start;i<end;i++)
{
it=find(vec.begin(),vec.end(),str[i]);
if(it!=vec.end())
return false;
else
vec.push_back(str[i]);
}
return true;
}
int lengthOfLongestSubstring(string s) {
int n=s.length();
int ans=0;
for(int i=0;i<n;i++)
{
for(int j=i+1;j<=n;j++)
{
if(CheckSbuStr(s,i,j))
ans=max(ans,j-i);
}
}
return ans;
}
};

滑动窗口法

  其实跟我所想的方法一模一样，但是表述的更加优雅。棒极了。使用容器自带的find函数得用set,vector中的find只能用迭代器查找，其实是algorithm中的find函数。当遇到相同元素时候，set中的容器就要删除到当前容器的前一元素。最大长度就是最大的i-left+1.
[TOC]
#include
#include
using namespace std;
class Solution {
public:
int lengthOfLongestSubstring(string s)
{
if(s.length()==0)return 0;
unordered_settemp;
int left=0;
int maxStr=0;
for(int i=0;i<s.length();i++)
{
while(temp.find(s[i])!=temp.end())
{
temp.erase(s[left]);
left++;
}
maxStr=max(maxStr,i-left+1);
temp.insert(s[i]);
}
return maxStr;
}
};

  滑动窗口也可以设立左指针a和右指针b，b指针向右伸缩，对每个元素判断是否出现过，若果出现过将a指针指向出现过的下一个位置，更新右指针和最大长度。
[TOC]

#include <iostream>
using namespace std;
class Solution {
    public:
    int lengthOfLongestSubstring(string s) 
    {
        int i(0),j(0),maxStr(0),len(0);
        while(j<s.length())
        {
            char tmpChar=s[j];
            for(int k=i;k<j;k++)
            {
                if(tmpChar==s[k])
                {
                    i=k+1;
                    len=j-i;
                    break;
                }
            }
            j++;
            len++;
            maxStr=max(maxStr,len);
        }
        return maxStr;
    }
};

  使用map优化查找是否出现过该元素。当判断是毕竟要判断是否出现过，还要判断这个出现过的元素的编号是不是大于i,即这个元素是不是在窗口里面。
[TOC]

class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {

        int i(0), j(0), len(0), maxStr(0);
        unordered_map<char, int> m;
        while (j<s.length())
        {
            char tmpChar = s[j];
            if (m.find(tmpChar) != m.end() && m[tmpChar] >= i)
            {
                i = m[tmpChar] + 1;
                len = j - i;
            }
            m[tmpChar] = j;

            j++;
            len++;
            maxStr = max(maxStr, len);
        }
        return maxStr;
    }
};
