LeetCode-合并两个有序数组
Created 2020-03-29| Updated 2020-03-28
Post View:

  首先使用排序函数偷个懒，或者直接拼接两个vector，然后用任意排序算法排序。本题本意应该是考双指针。

#include "iostream"
#include "vector"
#include "algorithm"
using namespace std;
class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        for(int i=0;i<n;++i)
        {
            nums1.push_back(nums2[i]);
        }
        sort(nums1.begin(),nums1.end());
        int cnt=0;
        int size=nums1.size();
        for(auto it=nums1.begin();it!=nums1.end(); )
        {
            if(*it==0 && cnt<size-m-n )
            {
                it=nums1.erase(it);
                cnt++;
            }
            else
            {
                ++it;
            }   
        }

    }
};

  其实不用计算0的个数，完全多此一举。直接把nums2接在nums1后即可。

#include "iostream"
#include "vector"
#include "algorithm"
using namespace std;

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        if(m==0)nums1=nums2;

        for(int i=m,j=0;i<m+n;i++,j++)
        {
            nums1[i]=nums2[j];
        }
        sort(nums1.begin(),nums1.end());
    }
};

  考虑双指针的方法，根据题目要求可以知道，nums1的空余元素是在尾部，所以在双指针从后遍历，比较得到较大的就放在后面的位置，当nums1的指针i遍历结束后且nums2的指针j没有结束，这证明nums2中的剩余元素均比nums1的元素小，直接加在nums1前面即可。还存在可以优化的地方，比如k的设计可以设计成一个指针，不断往前走。

#include "iostream"
#include "vector"
#include "algorithm"
using namespace std;

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {

        if(m==0)nums1=nums2;
        else if(n==0)nums1=nums1;
        else
        {
            int i,j,k;
            for( i=m-1,j=n-1,k=1; j>=0; )
            {
                if(nums1[i]<=nums2[j])
                {
                    nums1[n+m-k]=nums2[j];
                    j--;
                    k++;
                }
                else
                {
                    nums1[n+m-k]=nums1[i];
                    i--;
                    k++;
                    if(i<0)
                    {
                        while(j>=0)
                        {
                            nums1[j]=nums2[j];
                            j--;
                        }
                    }
                }
            }
        }
    }
};

  再就是利用一个新数组，双指针从前往后比较也可以，但是用到了额外空间。很简单，这里就不写了。
