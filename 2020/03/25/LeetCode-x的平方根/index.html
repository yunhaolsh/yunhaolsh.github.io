LeetCode-X的平方根
Created 2020-03-25| Updated 2020-03-25
Post View:

  最暴力的方法，用int会溢出，改成long后AC。当然这肯定是最傻的方法。

#include "iostream"
using namespace std;
class Solution {
public:
    long mySqrt(int x) {
        if(x==1)return 1;
        long i=1;
        while(i*i<=x)
        {
            i++;
        }

        return i-1;
    }
};

  改进一下先对半切查找一下，速度提升一丢丢，但是用了两个while循环，实际上比上面一个while的循环次数要少很多。

#include "iostream"
using namespace std;
class Solution {
public:
    long mySqrt(int x) {
        if(x==1)return 1;
        long n = x/2;

        while(n*n>x)
        {
            n/=2;
        }
        while(n*n<=x)
        {
            n++;
        }
        return n-1;
    }
};

  二分查找法很好用，要注意边界取值和更新。

class Solution {
public:
    int mySqrt(int x) {
        if(x==0 || x==1)return x;
        int mid,right=x,left=0;
        while(left<=right)
        {
            mid=left+(right-left)/2;
            if(mid==x/mid)return mid;
            else if((mid<x/mid) && ((mid+1)>x/(mid+1))) return mid;
            else if( mid<x/mid)
            {
                left=mid+1;
            }
            else
            {
                right=mid-1;
            }
        }
        return mid;
    }
};

官方题解 递归+位运算

本方法的思路是使用递归，每一步都减小x，直到 x < 2。当 x < 2 时有 sqrt{x} = x
​
无需减小。当 x>=2 时，减小方法如下：

mySqrt(x) = 2 *mySqrt(x/4)

例如：sqrt(x) = 2*sqrt(x/4)

减小 x 可以用左移或右移实现，因为这种方式计算效率非常高。

x << y 表示 x * 2^y

x >> y 表示 x*2^y

因此，递归式可以重写为：

mySqrt(x)=mySqrt(x>>2)<<1

class Solution {
      public int mySqrt(int x) {
    if (x < 2) return x;

    int left = mySqrt(x >> 2) << 1;
    int right = left + 1;
    return (long)right * right > x ? left : right;
      }
}
